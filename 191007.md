```python
def binary_word(n: int) -> str:
    """
    :type n: int
    :return: A string of binary digits which is 
    the binary representation of the integer n.
    :rtype: str    
    """
    word: str = ''
    while n _ _:
        word = str(n _ 2) + word
        n = n _ 2
    return _
```
